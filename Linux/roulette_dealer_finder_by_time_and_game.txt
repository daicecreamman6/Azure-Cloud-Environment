#!/bin/bash

#create new tmp file organized by game

cat $1 | sed s/" "/_/ | sed s/" "/_Blackjack_/ | awk -F" " '{print $1, $2}' >> dealer_by_game.tmp

cat $1 | sed s/" "/_/ | sed s/" "/_Blackjack_/ | sed s/" "/_Roulette_/ | awk -F" " '{print $1, $3}' >> dealer_by_game.tmp

cat $1 | sed s/" "/_/ | sed s/" "/_Blackjack_/ | sed s/" "/_Roulette_/ | sed s/" "/_TexasHoldEm_/ | awk -F" " '{print $1, $4}' >> dealer_by_game.tmp

#search new tmp file for time(accepting 4 digits)/AM or PM/Game (all one word)

cat dealer_by_game.tmp | grep $2':00*' | grep $3 | grep -i $4 | sed s/_/" "/ | sed s/_/" "/ | sed s/_/" "/ | awk -F" " '{print $3, $5}'

#remove tmp

rm dealer_by_game.tmp
